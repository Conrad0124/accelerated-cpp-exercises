# 指定最低版本要求
cmake_minimum_required(VERSION 3.10)
# 定义项目名称和所使用的语言,CXX代表C++
project(StudentGrades CXX)
# 指定C++标准并要求如果无法使用该标准则停止编译，不可以采用旧标准
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED TRUE)
 # 添加一个可执行文件目标-main3,其后是所有需要被编译的源文件
add_executable(main3
main3.cpp
median.cpp
grade.cpp
Student_info.cpp
)

# 说明上方两个命令：
# 第一条命令中的 .. 代表上一层目录，-DCMAKE_BUILD_TYPE=Debug代表
# 自动加上-g标志，保留debug信息,生成Makefiles
# -G选项：指定生成器，指定成MinGW的Makefiles做为生成器，不然
# 在安装了VisualStudio的电脑中默认生成器是使用VS的。
# 第二条命令中的 . 代表当前目录，生成.exe文件
